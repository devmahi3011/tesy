import customtkinter as ctk
import pyodbc
import bcrypt
import pandas as pd
from tkinter import ttk, messagebox

ctk.set_appearance_mode("dark")
ctk.set_default_color_theme("blue")

# -------- DB CONNECTION --------
def connect_db():
    return pyodbc.connect(
        'DRIVER={ODBC Driver 17 for SQL Server};'
        'SERVER=YOUR_SERVER;'
        'DATABASE=UserDB;'
        'UID=YOUR_USER;'
        'PWD=YOUR_PASSWORD'
    )

# -------- FETCH DATA --------
def fetch_data():
    conn = connect_db()
    query = "SELECT id, name, username, team, created_on, active FROM Users"
    df = pd.read_sql(query, conn)
    conn.close()
    return df

# -------- EXPORT TO EXCEL --------
def export_to_excel():
    df = fetch_data()
    df.to_excel("users_data.xlsx", index=False)
    messagebox.showinfo("Success", "Excel file created!")

# -------- SHOW TABLE --------
def show_table():
    df = fetch_data()

    table_window = ctk.CTkToplevel()
    table_window.title("User Data")
    table_window.geometry("800x400")

    tree = ttk.Treeview(table_window)
    tree.pack(fill="both", expand=True)

    tree["columns"] = list(df.columns)
    tree["show"] = "headings"

    for col in df.columns:
        tree.heading(col, text=col)
        tree.column(col, width=120)

    for _, row in df.iterrows():
        tree.insert("", "end", values=list(row))

# -------- DASHBOARD --------
def open_dashboard(username):
    dashboard = ctk.CTkToplevel()
    dashboard.title("Dashboard")
    dashboard.geometry("400x400")

    ctk.CTkLabel(dashboard, text=f"Welcome {username} 🎉", font=("Arial", 20)).pack(pady=20)

    ctk.CTkButton(dashboard, text="View Users", command=show_table).pack(pady=10)
    ctk.CTkButton(dashboard, text="Export to Excel", command=export_to_excel).pack(pady=10)

# -------- LOGIN --------
def login():
    username = login_user.get()
    password = login_pass.get()

    try:
        conn = connect_db()
        cursor = conn.cursor()

        cursor.execute("SELECT password FROM Users WHERE username=?", (username,))
        result = cursor.fetchone()

        if result and bcrypt.checkpw(password.encode(), result[0]):
            messagebox.showinfo("Success", "Login Successful 🎉")
            open_dashboard(username)   # 🔥 OPEN NEW WINDOW
        else:
            messagebox.showerror("Error", "Invalid credentials")

    except Exception as e:
        messagebox.showerror("Error", str(e))

# -------- REGISTER --------
def register():
    name = reg_name.get()
    username = reg_user.get()
    password = reg_pass.get()
    team = reg_team.get()

    if not (name and username and password and team):
        messagebox.showerror("Error", "All fields required")
        return

    hashed = bcrypt.hashpw(password.encode(), bcrypt.gensalt())

    try:
        conn = connect_db()
        cursor = conn.cursor()

        cursor.execute(
            "INSERT INTO Users (name, username, password, team, created_by) VALUES (?, ?, ?, ?, ?)",
            (name, username, hashed, team, "system")
        )
        conn.commit()

        messagebox.showinfo("Success", "Registered Successfully")
        show_login()

    except Exception as e:
        messagebox.showerror("Error", str(e))

# -------- SWITCH SCREENS --------
def show_register():
    login_frame.pack_forget()
    register_frame.pack(expand=True)

def show_login():
    register_frame.pack_forget()
    login_frame.pack(expand=True)

# -------- MAIN APP --------
app = ctk.CTk()
app.title("Login System")
app.geometry("400x500")

# -------- LOGIN FRAME --------
login_frame = ctk.CTkFrame(app)

ctk.CTkLabel(login_frame, text="Login", font=("Arial", 24)).pack(pady=20)

login_user = ctk.CTkEntry(login_frame, placeholder_text="Username")
login_user.pack(pady=10)

login_pass = ctk.CTkEntry(login_frame, placeholder_text="Password", show="*")
login_pass.pack(pady=10)

ctk.CTkButton(login_frame, text="Login", command=login).pack(pady=15)
ctk.CTkButton(login_frame, text="Create Account", command=show_register).pack()

login_frame.pack(expand=True)

# -------- REGISTER FRAME --------
register_frame = ctk.CTkFrame(app)

ctk.CTkLabel(register_frame, text="Register", font=("Arial", 24)).pack(pady=20)

reg_name = ctk.CTkEntry(register_frame, placeholder_text="Full Name")
reg_name.pack(pady=8)

reg_user = ctk.CTkEntry(register_frame, placeholder_text="Username")
reg_user.pack(pady=8)

reg_pass = ctk.CTkEntry(register_frame, placeholder_text="Password", show="*")
reg_pass.pack(pady=8)

reg_team = ctk.CTkEntry(register_frame, placeholder_text="Team")
reg_team.pack(pady=8)

ctk.CTkButton(register_frame, text="Register", command=register).pack(pady=15)
ctk.CTkButton(register_frame, text="Back to Login", command=show_login).pack()

# -------- RUN --------
app.mainloop()
