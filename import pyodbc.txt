def hash_password(password):
    return hashlib.sha256(password.encode()).hexdigest()


class LoginWindow:
    def __init__(self):
        self.root = ctk.CTk()
        self.root.title("Login")
        self.root.geometry("300x200")

        ctk.CTkLabel(self.root, text="Username").pack(pady=5)
        self.username = ctk.CTkEntry(self.root)
        self.username.pack(pady=5)

        ctk.CTkLabel(self.root, text="Password").pack(pady=5)
        self.password = ctk.CTkEntry(self.root, show="*")
        self.password.pack(pady=5)

        ctk.CTkButton(self.root, text="Login", command=self.login).pack(pady=15)

    def login(self):
        user = self.username.get()
        pwd = self.password.get()
        hashed_pwd = hash_password(pwd)

        try:
            conn = pyodbc.connect(
                "DRIVER={SQL Server};"
                "SERVER=TSTSERVER;"
                "DATABASE=DBTST;"
                "Trusted_Connection=yes;"
            )
            cursor = conn.cursor()
            cursor.execute(
                "SELECT * FROM Users WHERE username=? AND password=?",
                (user, hashed_pwd)
            )
            result = cursor.fetchone()
            conn.close()

            if result:
                self.root.destroy()
                main = ctk.CTk()
                app = AIOTApp(main)
                main.mainloop()
            else:
                messagebox.showerror("Error", "Invalid login")

        except Exception as e:
            messagebox.showerror("Error", f"Database error: {e}")

    def run(self):
        self.root.mainloop()

        """Called when pool dropdown changes"""
        self.load_sync_queue_options()

    def log(self, message, tag="info"):
        self.output.insert(tk.END, str(message) + "\n", tag)
        self.output.see(tk.END)
        self.root.update()

    def load_sync_queue_options(self):
        pool = self.pool_var.get()
        from_datetime = f"{self.from_date.get_date().strftime('%Y-%m-%d')} 00:00:00"
        to_datetime = f"{self.to_date.get_date().strftime('%Y-%m-%d')} 23:59:59"

        conn = None
        cursor = None
        try:
            conn = pyodbc.connect(
                "DRIVER={SQL Server};"
                "SERVER=TSTSERVER;"
                "DATABASE=DBTST;"
                "Trusted_Connection=yes;"
            )
            cursor = conn.cursor()

            with open("PROD_SQ.sql", "r") as file:
                query = file.read()

            cursor.execute(query, (pool, from_datetime, to_datetime))
            sync_groups = [row[0] for row in cursor.fetchall()]

            if not sync_groups:
                sync_groups = ["No Data Found"]

            self.sync_queue_dropdown.configure(values=sync_groups)
            self.sync_queue_dropdown.set(sync_groups[0])

        except Exception as e:
            self.log(f"Failed to load Sync Queue options: {e}", "error")
            self.sync_queue_dropdown.configure(values=["Error Loading Data"])
            self.sync_queue_dropdown.set("Error Loading Data")

        finally:
            if cursor:
                cursor.close()
            if conn:
                conn.close()

    def run_process(self):
        pool = self.pool_var.get()
        sync_queue_pattern = f"%{self.sync_queue_var.get()}%"
        from_datetime = f"{self.from_date.get_date().strftime('%Y-%m-%d')} 00:00:00"
        to_datetime = f"{self.to_date.get_date().strftime('%Y-%m-%d')} 23:59:59"

        # Clear previous
        for i in self.tree.get_children():
            self.tree.delete(i)
        self.reprocess_selected_btn.configure(state="disabled")
        self.selected_fg = None
        self.log("Running search...", "info")

        conn = None
        cursor = None
        try:
            conn = pyodbc.connect(
                "DRIVER={SQL Server};"
                "SERVER=TSTSERVER;"
                "DATABASE=DBTST;"
                "Trusted_Connection=yes;"
            )
            cursor = conn.cursor()

            with open("FailedFG.sql", "r") as f:
                query = f.read()

            cursor.execute(query, (pool, sync_queue_pattern, from_datetime, to_datetime))
            rows = cursor.fetchall()

            if not rows:
                self.log("No failed FG Serial Numbers found.", "error")
                return

            for row in rows:
                serial_number = row.FGSerialNumber if hasattr(row, "FGSerialNumber") else row[1]
                fail_datetime = row.FinishedOn if hasattr(row, "FinishedOn") else row[0]
                reason = row.Info if hasattr(row, "Info") else row[2]

                if fail_datetime:
                    fail_datetime = fail_datetime.strftime("%Y-%m-%d %H:%M:%S")

                self.tree.insert("", "end", values=(serial_number, fail_datetime, reason))

            self.log(f"Loaded {len(rows)} records.", "info")

        except Exception as e:
            messagebox.showerror("Error", str(e))
            self.log(f"Error: {e}", "error")

        finally:
            if cursor:
                cursor.close()
            if conn:
                conn.close()

    def on_row_select(self, event):
        selected = self.tree.selection()
        if selected:
            self.selected_fg = self.tree.item(selected[0])["values"][0]
            self.log(f"Selected: {self.selected_fg}", "serial")
            self.reprocess_selected_btn.configure(state="normal")
        else:
            self.selected_fg = None
            self.reprocess_selected_btn.configure(state="disabled")

    def process_fg(self, FGSerialNumber, cursor, session):
        pool = self.pool_var.get()
        sync_queue_pattern = f"%{self.sync_queue_var.get()}%"
        from_datetime = f"{self.from_date.get_date().strftime('%Y-%m-%d')} 00:00:00"
        to_datetime = f"{self.to_date.get_date().strftime('%Y-%m-%d')} 23:59:59"

        if not FGSerialNumber:
            self.log("Invalid Serial Number selected.", "error")
            return

        self.log(f"Processing: {FGSerialNumber}", "serial")

        try:
            with open("AIOT03TSTT.sql", "r") as f:
                query = f.read()

            cursor.execute(query, (
                pool,
                sync_queue_pattern,
                from_datetime,
                to_datetime,
                FGSerialNumber
            ))

            rows = cursor.fetchall()

            if not rows:
                self.log(f"No data found for Serial Number: {FGSerialNumber}", "error")
                return

            for row in rows:
                msg = row[0]
                payload = {
                    "Inputs": {"IncomingMessage": msg},
                    "ExecutionMode": 1,
                    "Retries": 1,
                    "SleepTime": 100,
                    "SynchronousQueue": "",
                    "Timeout": 100
                }

                try:
                    response = session.post(
                        AIOT03_DEV,
                        json=payload,
                        verify=False,
                        timeout=30
                    )

                    if response.status_code in (200, 201):
                        self.log(f"{FGSerialNumber} → POST Success", "success")
                        try:
                            self.log(f"Response: {response.json()}", "info")
                        except Exception:
                            self.log(f"Response Text: {response.text}", "info")
                    else:
                        self.log(f"{FGSerialNumber} → POST Failed: {response.status_code}", "error")

                except requests.exceptions.RequestException as req_err:
                    self.log(f"{FGSerialNumber} → Request Error: {req_err}", "error")

        except Exception as e:
            self.log(f"Error processing {FGSerialNumber}: {e}", "error")

    def reprocess_selected(self):
        if not self.selected_fg:
            messagebox.showwarning("Warning", "Select a Serial Number first")
            return

        conn = None
        cursor = None
        session = None
        try:
            conn = pyodbc.connect(
                "DRIVER={SQL Server};"
                "SERVER=TSTSERVER;"
                "DATABASE=DBTST;"
                "Trusted_Connection=yes;"
            )
            cursor = conn.cursor()
            session = requests.Session()

            self.process_fg(self.selected_fg, cursor, session)

        except Exception as e:
            self.log(f"Error: {e}", "error")

        finally:
            if session:
                session.close()
            if cursor:
                cursor.close()
            if conn:
                conn.close()
            self.log("Reprocess completed.", "info")

    def reprocess_all(self):
        items = self.tree.get_children()
        if not items:
            messagebox.showwarning("Warning", "No Serial Numbers to process")
            return

        self.log("Starting Reprocess All...", "info")

        conn = None
        cursor = None
        session = None
        try:
            conn = pyodbc.connect(
                "DRIVER={SQL Server};"
                "SERVER=TSTSERVER;"
                "DATABASE=DBTST;"
                "Trusted_Connection=yes;"
            )
            cursor = conn.cursor()
            session = requests.Session()

            for item in items:
                fg_serial = self.tree.item(item)["values"][0]
                self.tree.selection_set(item)
                self.tree.focus(item)
                self.root.update()

                self.process_fg(fg_serial, cursor, session)

        except Exception as e:
            self.log(f"Error: {e}", "error")

        finally:
            if session:
                session.close()
            if cursor:
                cursor.close()
            if conn:
                conn.close()

        self.log("All Serial Numbers processed.", "success")

    def export_excel(self):
        """Export the failed list to Excel"""
        items = self.tree.get_children()
        if not items:
            messagebox.showwarning("Warning", "No data to export")
            return

        data = []
        for item in items:
            values = self.tree.item(item)["values"]
            data.append({
                "Serial Number": values[0],
                "Fail DateTime": values[1],
                "Reason": values[2]
            })

        df = pd.DataFrame(data)

        # Generate filename with timestamp
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        filename = f"failed_fg_export_{timestamp}.xlsx"

        try:
            df.to_excel(filename, index=False)
            self.log(f"Exported to {filename}", "success")
            messagebox.showinfo("Success", f"Data exported to {filename}")
        except Exception as e:
            self.log(f"Export failed: {e}", "error")
            messagebox.showerror("Error", f"Export failed: {e}")
